# Number-base-conversion
dec = 1991 #เลขฐาน10
conv_to = bin(dec) #แปลงเลขฐาน10 to 2
dec = (conv_to)
print(dec)
 
dec2 = 1991 #เลขฐาน10
conv_to2 = oct(dec2) #แปลงเลขฐาน10 to 8
dec2 = (conv_to2)
print(dec2)

dec3 = 1991 #เลขฐาน10
conv_to3 = hex(dec3) #แปลงเลขฐาน10 to 16    
dec3 = (conv_to3)
print(dec3)
